                        newActive.classList.add('opacity-100', 'z-10');

                        newActive.classList.remove('opacity-0', 'z-0');

                    }

                    

                    currentReview = index;

                }


                function next() {

                    const nextIndex = (currentReview + 1) % reviews.length;

                    showReview(nextIndex);

                    resetTimer();

                }


                function prev() {

                    const prevIndex = (currentReview - 1 + reviews.length) % reviews.length;

                    showReview(prevIndex);

                    resetTimer();

                }


                function resetTimer() {

                    clearInterval(autoSlideTimer);

                    autoSlideTimer = setInterval(next, 7000);

                }


                if (reviews.length > 1) {

                    nextBtn.addEventListener('click', next);

                    prevBtn.addEventListener('click', prev);

                    resetTimer(); // Start the timer

                }

            }
